CREATE TABLE users(
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL,
    hash TEXT NOT NULL,
    room_id INTEGER, 
    FOREIGN KEY(room_id) REFERENCES rooms(id)
);
CREATE TABLE rooms(
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    admin_id INTEGER NOT NULL,
    room_id INTEGER NOT NULL,
    members INTEGER NOT NULL,
    FOREIGN KEY(admin_id) REFERENCES users(id)
);
